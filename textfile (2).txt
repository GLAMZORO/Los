local RebirthTab = Window:Tab({Title = "Auto Rebirth", Icon = "refresh-cw", Locked = false})
local autoRebirth = false
local rebirthTask

RebirthTab:Toggle({
    Title = "Auto Rebirth", 
    Desc = "Automatically spam rebirth", 
    Default = false, 
    Callback = function(state)
        autoRebirth = state
        if state then
            rebirthTask = task.spawn(function()
                while autoRebirth do
                    game:GetService("ReplicatedStorage").rEvents.rebirthEvent:FireServer("rebirthRequest")
                    task.wait(0.1)
                end
            end)
        else
            if rebirthTask then
                task.cancel(rebirthTask)
            end
        end
    end
})

RebirthTab:Button({
    Title = "Stop Auto Rebirth", 
    Desc = "Stop auto rebirth immediately", 
    ButtonIcon = "x-circle", 
    Callback = function()
        autoRebirth = false
        if rebirthTask then
            task.cancel(rebirthTask)
        end
        print("Stopped auto rebirth")
    end
})
